            public interface IFooBarzable {
                public function foo(... pairs):Array;
            }

            public class FooBar implements IFooBarzable {
                static private var cnt:uint = 0;
                private var bar:String;

                //constructor
                public function TestBar(bar:String):void {
                    bar = bar;
                    ++cnt;
                }

                public function foo(... pairs):Array {
                    pairs.push(bar);
                    return pairs;
                }
            }
